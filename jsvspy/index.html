For ciklus

JavaScript Python 

// klasszikus for
for (let i = 0; i < 5; i++) {
  console.log(i);
}

// for-of
let lista = [10,20,30];
for (let elem of lista) {
  console.log(elem);
}
              

# range alapú
for i in range(5):
    print(i)

# lista bejárása
lista = [10,20,30]
for elem in lista:
    print(elem)
